include(WiresharkPlugin)

# Plugin name and version info (major minor micro extra)
set_module_info(proto1 0 0 1 0)

set(DISSECTOR_SRC
	packet-proto1.c
)

set(PLUGIN_FILES
	plugin.c
	${DISSECTOR_SRC}
)

set_source_files_properties(
	${PLUGIN_FILES}
	PROPERTIES
	COMPILE_FLAGS "${WERROR_COMMON_FLAGS}"
)

register_plugin_files(plugin.c
	plugin
	${DISSECTOR_SRC}
)

add_plugin_library(proto1 epan)

target_link_libraries(proto1 epan ${MRUBY_LIBRARIES})
target_include_directories(proto1 PRIVATE ${MRUBY_INCLUDE_DIRS})

install_plugin(proto1 epan)

file(GLOB DISSECTOR_HEADERS RELATIVE "${CMAKE_CURRENT_SOURCE_DIR}" "*.h")
CHECKAPI(
	NAME
	  proto1
	SWITCHES
	  --group dissectors-prohibited
	  --group dissectors-restricted
	SOURCES
	  ${DISSECTOR_SRC}
	  ${DISSECTOR_HEADERS}
)
